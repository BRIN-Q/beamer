\begin{frame}{Algorithmic Pseudocode}
%-----------------------------------------------------------
    \begin{algorithm}[H]
        \caption{Euclidean Distance}
        \begin{algorithmic}[1]
            \State \textbf{Input:} Points $P$ and $Q$
            \State $sum \leftarrow 0$
            \For{$i = 1$ to $n$}
                \State $diff \leftarrow P_i - Q_i$
                \State $sum \leftarrow sum + diff^2$
            \EndFor
            \State \Return $\sqrt{sum}$
        \end{algorithmic}
    \end{algorithm}
\end{frame}
